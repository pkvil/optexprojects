\setbox\guidebox\vbox{%
\directmetapost{%
beginfig(1);
	% Variables (settings)
	numeric nib,a,b,d;
	nib=2.4mm; 			% nib width
	a=3; 				% ascender height in nibs
	b=4; 				% body height in nibs
	d=3; 				% descender depth in nibs
	
	% Pens
	pen fineliner;
	fineliner = pencircle scaled 0.2pt;
	pickup fineliner;
	
	% Coordinates (origin at bottom left corner of body square)
	pair A[],B[],D[]; 			% Ascender points, Body points, Descender points 
	A[0]:=( 0, (a+b)*nib ); 	% left point
	A[1]:=( b*nib, (a+b)*nib ); % right point
	B[0]:=( origin ); 			% southwest corner
	B[1]:=( 0, b*nib ); 		% northwest corner
	B[2]:=( b*nib, b*nib ); 	% northeast corner
	B[3]:=( b*nib, 0); 			% southeast corner
	D[0]:=( 0, -d*nib ); 		% left point
	D[1]:=( b*nib, -d*nib ); 	% right point
	
	% Paths
	path square, circle, ascenderline, descenderline, leftnarrower, rightnarrower;
	square = B[0]--B[1]--B[2]--B[3]--cycle;
	circle = fullcircle scaled (b*nib) shifted (0.5*b*nib,0.5*b*nib);
	ascenderline = A[0]--A[1];
	descenderline = D[0]--D[1];
	leftnarrower = 1/8[A[0],A[1]] -- 1/8[D[0],D[1]];
	rightnarrower = 7/8[A[0],A[1]] -- 7/8[D[0],D[1]];
	
	pair T[]; % Thirty degree guidelines derived from circle
	T[0]:= point 2/3 of circle;
	T[1]:= point 4+2/3 of circle;
	T[2]:= point 2+2/3 of circle;
	T[3]:= point 6+2/3 of circle;
	
	% Draw
	
	draw ascenderline;
	draw square;
	draw circle ;
	draw leftnarrower;
	draw rightnarrower;
	draw descenderline;
	draw  T[0] -- T[1];
	draw  T[2] -- T[3];

endfig;
end;
}%
}
