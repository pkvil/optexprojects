%
% My macros
%

% \skiptorecto is a page break that skips to a recto (right side) page
\def\skiptorecto{%
    \vfil\break
    \ifodd\pageno\else\vfil\break\fi
}

% let \act be a synonym for \secc
\let\act=\secc

% set \seccfont to normal size and normal weight
\_def \_seccfont {\_scalemain\_typoscale[\_magstep0/\_magstep0]\rm}

% set \printsecc to insert space above, but not at the top of a page. Print the secc name (argument of \secc) in a centered line. 
\_def\_printsecc#1{\_par
   \_abovetitle{\_penalty-200}{\vskip2\baselineskip}
   {\_seccfont \centerline{#1}}%
   \_nobreak \_belowtitle{\vskip2\baselineskip}%
   \_firstnoindent
}
