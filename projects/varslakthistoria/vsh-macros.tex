\def\startblockquote{%
	\par
	\vskip\baselineskip
	\it
	\advance\leftskip by 2\parindent
	\rightskip=\leftskip
	\_firstnoindent
}

\def\stopblockquote{%
	\par
	\vskip\baselineskip
	\rm
	\advance\leftskip by -2\parindent
	\rightskip=\leftskip
	\_firstnoindent
}

%%
%% Sectioning
%%

\def\person#1{%
	\vfill\break
	\ifodd\pageno
		\else \null \vfill\break
	\fi
	{\typosize[16/]\currvar
	 \smash{\centerline{#1}}%
	 \vskip4\baselineskip
	 \_firstnoindent
	}%
}

\def\couple#1#2{%
	\vfill\break
	\ifodd\pageno
		\else \null \vfill\break
	\fi
	{\typosize[15/]\currvar 
	 \smash{\centerline{#1}}%
	 \vskip0.5\baselineskip
	 \smash{\centerline{\&}}%
	 \vskip0.5\baselineskip
	 \smash{\centerline{#2}}%
	 \vskip4\baselineskip
	 \_firstnoindent
	}%
}
