\def\startblockquote{%
	\par
	\vskip\baselineskip
	\it
	\advance\leftskip by 2\parindent
	\rightskip=\leftskip
	\_firstnoindent
}

\def\stopblockquote{%
	\par
	\vskip\baselineskip
	\rm
	\advance\leftskip by -2\parindent
	\rightskip=\leftskip
	\_firstnoindent
}

%%
%% Sectioning
%%

\def\person#1{%
	\vfill\break
	\ifodd\pageno
		\else \null \vfill\break
	\fi
	{\typosize[18/]\subhlight\currvar \smash{\centerline{#1}}}
	\vskip1\baselineskip
	\centerline{\smash{\typosize[14/]\lower1ex\hbox{\myred \downthreeleaf}}}
	\vskip2\baselineskip
	\_firstnoindent
}

\def\couple#1#2{%
	\vfill\break
	\ifodd\pageno
		\else \null \vfill\break
	\fi
	{\typosize[18/]\subh\currvar 
	 \smash{\centerline{#1}}%
	\vskip0.5\baselineskip
	\smash{\centerline{\it \&}}%
	\vskip0.5\baselineskip
	\smash{\centerline{#2}}}%
	\vskip1\baselineskip
	\centerline{\smash{\typosize[14/]\lower1ex\hbox{\Black \downthreeleaf}}}
	\vskip2\baselineskip
	\_firstnoindent
	
}
