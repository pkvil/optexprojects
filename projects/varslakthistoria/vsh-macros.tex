% Block quotes

\def\startblockquote{%
	\par
	\vskip\baselineskip
	\advance\leftskip by 2\parindent
	\rightskip=\leftskip
	\_firstnoindent
	\it
}

	\par
	\vskip\baselineskip
	\rm
	\advance\leftskip by -2\parindent
	\rightskip=\leftskip
	\_firstnoindent
}

\def\part#1{%
	\skiptorecto
	\vglue 4\baselineskip
	\footlinefalse
	{\typosize[20/]\setff{+titl,letterspace=10}\rm \smash{\centerline{#1}}}
}

\def\person#1#2{%
	\skiptorecto
	{\typosize[16/]\setff{+smcp,letterspace=10}\rm 
		%\fontdimen2\font=45pt\relax
		\smash{\centerline{#1}}%
	}%
	\vskip1\baselineskip
	\centerline{\it #2}
	\vskip4\baselineskip
	\_firstnoindent
}

\def\couple#1#2{%
	\skiptorecto
	{\typosize[16/]\rm%
		%\the\fontdimen2\font
		%\fontdimen2\font=4.5pt\relax
		\smash{\centerline{\setff{+smcp, letterspace=10}\currvar #1}}%
		\vskip0.5\baselineskip
		\smash{\centerline{\it \&}}%
		\vskip0.5\baselineskip
		\smash{\centerline{\setff{+smcp, letterspace=10}\currvar #2}}%
	}%
	\vskip4\baselineskip
	\_firstnoindent
}

\def\section#1{%
	\vskip\baselineskip
	\noindent{\setff{+smcp,+c2sc, letterspace=10}\currvar #1\nbpar}
	\nobreak\vskip\baselineskip
	\_firstnoindent
}

\def\skiptorecto{%
	\vfill\break
	\ifodd\pageno
		\else \null\footlinefalse\vfill\break
	\fi
}

\def\putpic#1#2{%
	\vskip\baselineskip
	\vbox to 0pt{%
		\kern-1ex 
		\picheight=\dimexpr(#1\baselineskip+1ex)%
		\centerline{\inspic{#2}}%
	}%
	\vskip#1\baselineskip
	\vskip\baselineskip
	\_firstnoindent
}

\newdimen\vsizedefault
\newdimen\footlinedistdefault
\def\extravsizes#1{%
   \foreach #1\do ##1:##2{%
      \tmpnum=##1
      \sxdef{vs:\the\tmpnum}{##2}
      \advance\tmpnum by \ifodd\tmpnum -\fi 1
      \sxdef{vs:\the\tmpnum}{##2}
   }
   \global\vsizedefault=\vsize   % for page 1
   \global\footlinedistdefault=\footlinedist
   \corrvsize
   \global\pgbottomskip=0pt plus1fil minus\baselineskip
}
\addto\_begoutput{\global\vsize=\vsizedefault}
\addto\_endoutput{\corrvsize}

\def\corrvsize{
   \global\vsize=\vsizedefault
   \global\footlinedist=\footlinedistdefault
   \ifcsname vs:\the\pageno\endcsname
      \global\advance \vsize by \csname vs:\the\pageno\endcsname \baselineskip
      \global\advance \footlinedist by \csname vs:\the\pageno\endcsname \baselineskip
      
   \fi
}

\extravsizes{ }
