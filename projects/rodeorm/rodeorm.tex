		\def\initialcolor{\Red}%
		\bluefalse
	\else
		\def\initialcolor{\Blue}%
		\bluetrue
	\fi
}

	\dimen1=\dimexpr(\hsize-\dimen0)\relax

	\parshape 3 \dimen0 \dimen1 \dimen0 \dimen1 0pt \hsize
	\noindent
}
